async function predict() {
    const fileInput = document.getElementById("imageInput");
    const resultDiv = document.getElementById("result");

    if (!fileInput.files[0]) {
        alert("Iltimos rasm yuklang!");
        return;
    }

    resultDiv.innerHTML = "Tekshirilmoqda...";

    const file = fileInput.files[0];
    const reader = new FileReader();

    reader.onloadend = async function () {
        const base64 = reader.result.split(",")[1];

        const response = await fetch("https://serverless.roboflow.com/tiu-hwuau/workflows/custom-workflow", {
            method: "POST",
            headers: {
                "Content-Type": "application/json"
            },
            body: JSON.stringify({
                api_key: "api_key": "cf29DAu7zROhHc0V8FxS",
                inputs: {
                    image: {
                        type: "base64",
                        value: base64
                    }
                }
            })
        });

        const data = await response.json();
        console.log(data);

        if (data.outputs && data.outputs[0].predictions.length > 0) {
            const prediction = data.outputs[0].predictions[0];
            resultDiv.innerHTML =
                "Kasallik: " + prediction.class +
                "<br>Ishonchlilik: " + (prediction.confidence * 100).toFixed(2) + "%";
        } else {
            resultDiv.innerHTML = "Natija topilmadi.";
        }
    };

    reader.readAsDataURL(file);
}

document.getElementById("imageInput").addEventListener("change", function(e) {
    const preview = document.getElementById("preview");
    preview.src = URL.createObjectURL(e.target.files[0]);
});
